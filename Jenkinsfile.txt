pipeline {
    agent any

    stages {
        stage('Maven Clean') {
            steps {
                bat "mvn clean package -DskipTests"
            }
        }

        stage('Build Docker Image') {
            steps {
                bat "docker build -t=javabase ."
            }
        }

        stage('Push Docker Image') {
            steps {
                 bat "docker push javabase"
            }
        }
    }
}
